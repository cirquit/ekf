247  void apply_prediction()
248  {
249  // Using some zero cost abstraction renaming for mathematical understanding
250  const nxn_matrix & P = _prediction_error;
251  const nxn_matrix & Q = _process_noise;
252  const nxn_matrix & F = _f.jacobian(_state, _f_jacobian_temp);
253 
254  _prediction_error = F * P * blaze::trans(F) + Q;
255  _f(_state, _state);
256  _prediction_count++;
257  }
258 
273  void apply_update()
274  {
275  // Using zero cost abstraction renaming for mathematical understanding
276  _h(_state, _h_temp);
277  const mx1_vector & h = _h_temp;
278  const mxn_matrix & H = _h.jacobian(_state, _h_jacobian_temp);
279  const nxn_matrix & P = _prediction_error;
280  const mxm_matrix & cN = _sensor_noise;
281  const nxn_matrix & I = _identity;
282  const nx1_vector & s = _state;
283  const mx1_vector & o = _observation;
284  nxm_matrix & G = _gain;
285 
286  _gain = P * blaze::trans(H) * blaze::inv(H * P * blaze::trans(H) + cN);
287  _state = s + G * (o - h);
288  _prediction_error = (I - G * H) * P;
289  _update_count++;
290  }
